15 // This is a Basic algorithm (no extra cached properties are required),
16 // but it is not user-callable (see LAGr_BreadthFirstSearch instead).
17 
18 #define LG_FREE_WORK \
19 { \
20  GrB_free (&frontier); \
21 }
22 
23 #define LG_FREE_ALL \
24 { \
25  LG_FREE_WORK ; \
26  GrB_free (&l_parent); \
27  GrB_free (&l_level); \
28 }

31 
32 int LG_BreadthFirstSearch_vanilla
33 (
34  GrB_Vector *level,
35  GrB_Vector *parent,
36  const LAGraph_Graph G,
37  GrB_Index src,
38  char *msg
39 )
40 {
41 
42  //--------------------------------------------------------------------------
43  // check inputs
44  //--------------------------------------------------------------------------
45 
46  LG_CLEAR_MSG ;
47  GrB_Vector frontier = NULL; // the current frontier
48  GrB_Vector l_parent = NULL; // parent vector
49  GrB_Vector l_level = NULL; // level vector
50 
51  bool compute_level = (level != NULL);
52  bool compute_parent = (parent != NULL);
53  if (compute_level ) (*level ) = NULL;
54  if (compute_parent) (*parent) = NULL;
55  LG_ASSERT_MSG (compute_level || compute_parent, GrB_NULL_POINTER,
56  "either level or parent must be non-NULL") ;
57 
58  LG_TRY (LAGraph_CheckGraph (G, msg)) ;
59 
60  //--------------------------------------------------------------------------
61  // get the problem size
62  //--------------------------------------------------------------------------
63 
64  GrB_Matrix A = G->A ;
65 
66  GrB_Index n;
67  GRB_TRY( GrB_Matrix_nrows (&n, A) );
68  LG_ASSERT_MSG (src < n, GrB_INVALID_INDEX, "invalid source node") ;
69 
70  // determine the semiring type
71  GrB_Type int_type = (n > INT32_MAX) ? GrB_INT64 : GrB_INT32 ;
72  GrB_BinaryOp
73  second_op = (n > INT32_MAX) ? GrB_SECOND_INT64 : GrB_SECOND_INT32 ;
74  GrB_Semiring semiring = NULL;
75  GrB_IndexUnaryOp ramp = NULL ;
76 
77  if (compute_parent)
78  {
79  // create the parent vector. l_parent(i) is the parent id of node i
80  GRB_TRY (GrB_Vector_new(&l_parent, int_type, n)) ;
81 
82  semiring = (n > INT32_MAX) ?
83  GrB_MIN_FIRST_SEMIRING_INT64 : GrB_MIN_FIRST_SEMIRING_INT32;
84 
85  // create a sparse integer vector frontier, and set frontier(src) = src
86  GRB_TRY (GrB_Vector_new(&frontier, int_type, n)) ;
87  GRB_TRY (GrB_Vector_setElement(frontier, src, src)) ;
88 
89  // pick the ramp operator
90  ramp = (n > INT32_MAX) ? GrB_ROWINDEX_INT64 : GrB_ROWINDEX_INT32 ;
91  }
92  else
93  {
94  // only the level is needed
95  semiring = LAGraph_any_one_bool ;
96 
97  // create a sparse boolean vector frontier, and set frontier(src) = true
98  GRB_TRY (GrB_Vector_new(&frontier, GrB_BOOL, n)) ;
99  GRB_TRY (GrB_Vector_setElement(frontier, true, src)) ;
100  }
101 
102  if (compute_level)
103  {
104  // create the level vector. v(i) is the level of node i
105  // v (src) = 0 denotes the source node
106  GRB_TRY (GrB_Vector_new(&l_level, int_type, n)) ;
107  //GRB_TRY (GrB_Vector_setElement(l_level, 0, src)) ;
108  }
109 
110  //--------------------------------------------------------------------------
111  // BFS traversal and label the nodes
112  //--------------------------------------------------------------------------
113  GrB_Index nq = 1 ; // number of nodes in the current level
114  GrB_Index last_nq = 0 ;
115  GrB_Index current_level = 0;
116  GrB_Index nvals = 1;
117 
118  // {!mask} is the set of unvisited nodes
119  GrB_Vector mask = (compute_parent) ? l_parent : l_level ;
120 
121  // parent BFS
122  do
123  {
124  if (compute_level)
125  {
126  // assign levels: l_level<s(frontier)> = current_level
127  GRB_TRY( GrB_assign(l_level, frontier, GrB_NULL,
128  current_level, GrB_ALL, n, GrB_DESC_S) );
129  ++current_level;
130  }
131 
132  if (compute_parent)
133  {
134  // frontier(i) currently contains the parent id of node i in tree.
135  // l_parent<s(frontier)> = frontier
136  GRB_TRY( GrB_assign(l_parent, frontier, GrB_NULL,
137  frontier, GrB_ALL, n, GrB_DESC_S) );
138 
139  // convert all stored values in frontier to their indices
140  GRB_TRY (GrB_apply (frontier, GrB_NULL, GrB_NULL, ramp,
141  frontier, 0, GrB_NULL)) ;
142  }
143 
144  // frontier = kth level of the BFS
145  // mask is l_parent if computing parent, l_level if computing just level
146  GRB_TRY( GrB_vxm(frontier, mask, GrB_NULL, semiring,
147  frontier, A, GrB_DESC_RSC) );
148 
149  // done if frontier is empty
150  GRB_TRY( GrB_Vector_nvals(&nvals, frontier) );
151  } while (nvals > 0);
152 
153  //--------------------------------------------------------------------------
154  // free workspace and return result
155  //--------------------------------------------------------------------------
156 
157  if (compute_parent) (*parent) = l_parent ;
158  if (compute_level ) (*level ) = l_level ;
159  LG_FREE_WORK ;
160  return (GrB_SUCCESS) ;
161 }
