14 // Also computes G->AT if not already computed, if G is not an undirected
15 // graph and G->A is square.

17 #define LG_FREE_WORK \
18 { \
19  GrB_free (&S1) ; \
20  GrB_free (&S2) ; \
21  GrB_free (&C) ; \
22 }

24 #include "LG_internal.h"
25 
26 int LAGraph_Cached_IsSymmetricStructure
27 (
28  // input/output:
29  LAGraph_Graph G, // graph to determine the symmetry of structure of A
30  char *msg
31 )
32 {
33 
34  //--------------------------------------------------------------------------
35  // clear msg and check G
36  //--------------------------------------------------------------------------
37 
38  GrB_Matrix C = NULL, S1 = NULL, S2 = NULL ;
39  LG_CLEAR_MSG_AND_BASIC_ASSERT (G, msg) ;
40 
41  if (G->kind == LAGraph_ADJACENCY_UNDIRECTED)
42  {
43  // assume A is symmetric for an undirected graph
44  G->is_symmetric_structure = LAGraph_TRUE ;
45  return (GrB_SUCCESS) ;
46  }
47 
48  if (G->is_symmetric_structure != LAGRAPH_UNKNOWN)
49  {
50  // cached symmetric property is already known
51  return (GrB_SUCCESS) ;
52  }
53 
54  //--------------------------------------------------------------------------
55  // determine the size of A
56  //--------------------------------------------------------------------------
57 
58  GrB_Matrix A = G->A ;
59  GrB_Index n, ncols ;
60  GRB_TRY (GrB_Matrix_nrows (&n, A)) ;
61  GRB_TRY (GrB_Matrix_ncols (&ncols, A)) ;
62  if (n != ncols)
63  {
64  // A is rectangular and thus cannot be symmetric
65  G->is_symmetric_structure = LAGraph_FALSE ;
66  return (GrB_SUCCESS) ;
67  }
68 
69  //--------------------------------------------------------------------------
70  // compute the transpose, if not already computed
71  //--------------------------------------------------------------------------
72 
73  if (G->AT == NULL)
74  {
75  LG_TRY (LAGraph_Cached_AT (G, msg)) ;
76  }
77 
78  //--------------------------------------------------------------------------
79  // check if the structure of A and AT are the same
80  //--------------------------------------------------------------------------
81 
82  GRB_TRY (GrB_Matrix_new (&C, GrB_BOOL, n, n)) ;
83 
84  // C(i,j) = 1 if both A(i,j) and AT(i,j) exist
85  GRB_TRY (GrB_eWiseMult (C, NULL, NULL, GrB_ONEB_BOOL, A, G->AT, NULL)) ;
86 
87  GrB_Index nvals1, nvals2 ;
88  GRB_TRY (GrB_Matrix_nvals (&nvals1, C)) ;
89  GRB_TRY (GrB_Matrix_nvals (&nvals2, A)) ;
90  G->is_symmetric_structure =
91  (nvals1 == nvals2) ? LAGraph_TRUE : LAGraph_FALSE ;
92 
93  //--------------------------------------------------------------------------
94  // free workspace and return result
95  //--------------------------------------------------------------------------
96 
97  LG_FREE_WORK ;
98  return (GrB_SUCCESS) ;
99 }
