14 // Count the number of triangles in a graph,
15 
16 // This is an Advanced algorithm (G->nself_edges, G->out_degree,
17 // G->is_symmetric_structure are required).
18 
19 // Given a symmetric graph A with no-self edges, LAGr_TriangleCount counts the
20 // number of triangles in the graph. A triangle is a clique of size three,
21 // that is, 3 nodes that are all pairwise connected.
22 
23 // One of 6 methods are used, defined below where L and U are the strictly
24 // lower and strictly upper triangular parts of the symmetrix matrix A,
25 // respectively. Each method computes the same result, ntri:
26 
27 // 0: default: use the default method (currently methood 5)
28 // 1: Burkhardt: ntri = sum (sum ((A^2) .* A)) / 6
29 // 2: Cohen: ntri = sum (sum ((L * U) .* A)) / 2
30 // 3: Sandia: ntri = sum (sum ((L * L) .* L))
31 // 4: Sandia2: ntri = sum (sum ((U * U) .* U))
32 // 5: SandiaDot: ntri = sum (sum ((L * U') .* L)). Note that L=U'.
33 // 6: SandiaDot2: ntri = sum (sum ((U * L') .* U)). Note that U=L'.
34 
35 // A is a square symmetric matrix, of any type. Its values are ignored.
36 // Results are undefined for methods 1 and 2 if self-edges exist in A. Results
37 // are undefined for all methods if A is unsymmetric.
38 
39 // The Sandia* methods all tend to be faster than the Burkhardt or Cohen
40 // methods. For the largest graphs, SandiaDot tends to be fastest, except for
41 // the GAP-urand matrix, where the saxpy-based Sandia method (L*L.*L) is
42 // fastest. For many small graphs, the saxpy-based Sandia and Sandia2 methods
43 // are often faster that the dot-product-based methods.
44 
45 // Reference (for the "Sandia*" methods): Wolf, Deveci, Berry, Hammond,
46 // Rajamanickam, 'Fast linear algebra- based triangle counting with
47 // KokkosKernels', IEEE HPEC'17, https://dx.doi.org/10.1109/HPEC.2017.8091043,
48

61 static int tricount_prep
62 (
63  GrB_Matrix *L, // if present, compute L = tril (A,-1)
64  GrB_Matrix *U, // if present, compute U = triu (A, 1)
65  GrB_Matrix A, // input matrix
66  char *msg
67 )
68 {
69  GrB_Index n ;
70  GRB_TRY (GrB_Matrix_nrows (&n, A)) ;
71 
72  if (L != NULL)
73  {
74  // L = tril (A,-1)
75  GRB_TRY (GrB_Matrix_new (L, GrB_BOOL, n, n)) ;
76  GRB_TRY (GrB_select (*L, NULL, NULL, GrB_TRIL, A, (int64_t) (-1),
77  NULL)) ;
78  GRB_TRY (GrB_Matrix_wait (*L, GrB_MATERIALIZE)) ;
79  }
80 
81  if (U != NULL)
82  {
83  // U = triu (A,1)
84  GRB_TRY (GrB_Matrix_new (U, GrB_BOOL, n, n)) ;
85  GRB_TRY (GrB_select (*U, NULL, NULL, GrB_TRIU, A, (int64_t) 1, NULL)) ;
86  GRB_TRY (GrB_Matrix_wait (*U, GrB_MATERIALIZE)) ;
87  }
88  return (GrB_SUCCESS) ;
89 }

105 int LAGr_TriangleCount
106 (
107  // output:
108  uint64_t *ntriangles,
109  // input:
110  const LAGraph_Graph G,
111  LAGraph_TriangleCount_Method method,
112  LAGraph_TriangleCount_Presort *presort,
113  char *msg
114 )
115 {
116 
117  //--------------------------------------------------------------------------
118  // check inputs
119  //--------------------------------------------------------------------------
120 
121  LG_CLEAR_MSG ;
122  GrB_Matrix C = NULL, L = NULL, U = NULL, T = NULL ;
123  int64_t *P = NULL ;
124  LG_ASSERT_MSG (
125  method == LAGraph_TriangleCount_Default || // 0: use default method
126  method == LAGraph_TriangleCount_Burkhardt || // 1: sum (sum ((A^2) .* A))/6
127  method == LAGraph_TriangleCount_Cohen || // 2: sum (sum ((L * U) .*A))/2
128  method == LAGraph_TriangleCount_Sandia || // 3: sum (sum ((L * L) .* L))
129  method == LAGraph_TriangleCount_Sandia2 || // 4: sum (sum ((U * U) .* U))
130  method == LAGraph_TriangleCount_SandiaDot || // 5: sum (sum ((L * U') .* L))
131  method == LAGraph_TriangleCount_SandiaDot2, // 6: sum (sum ((U * L') .* U))
132  GrB_INVALID_VALUE, "method is invalid") ;
133  if (presort != NULL)
134  {
135  LG_ASSERT_MSG (
136  (*presort) == LAGraph_TriangleCount_NoSort ||
137  (*presort) == LAGraph_TriangleCount_Ascending ||
138  (*presort) == LAGraph_TriangleCount_Descending ||
139  (*presort) == LAGraph_TriangleCount_AutoSort,
140  GrB_INVALID_VALUE, "presort is invalid") ;
141  }
142  LG_TRY (LAGraph_CheckGraph (G, msg)) ;
143  LG_ASSERT (ntriangles != NULL, GrB_NULL_POINTER) ;
144  LG_ASSERT (G->nself_edges == 0, LAGRAPH_NO_SELF_EDGES_ALLOWED) ;
145 
146  if (method == LAGraph_TriangleCount_Default)
147  {
148  // 0: use default method (5): SandiaDot: sum (sum ((L * U') .* L))
149  method = LAGraph_TriangleCount_SandiaDot ;
150  }
151 
152  LG_ASSERT_MSG ((G->kind == LAGraph_ADJACENCY_UNDIRECTED ||
153  (G->kind == LAGraph_ADJACENCY_DIRECTED &&
154  G->is_symmetric_structure == LAGraph_TRUE)),
155  LAGRAPH_SYMMETRIC_STRUCTURE_REQUIRED,
156  "G->A must be known to be symmetric") ;
157 
158  // the Sandia* methods can benefit from the presort
159  bool method_can_use_presort =
160  method == LAGraph_TriangleCount_Sandia || // 3: sum (sum ((L * L) .* L))
161  method == LAGraph_TriangleCount_Sandia2 || // 4: sum (sum ((U * U) .* U))
162  method == LAGraph_TriangleCount_SandiaDot || // 5: sum (sum ((L * U') .* L))
163  method == LAGraph_TriangleCount_SandiaDot2 ; // 6: sum (sum ((U * L') .* U))
164 
165  GrB_Matrix A = G->A ;
166  GrB_Vector Degree = G->out_degree ;
167  bool auto_sort = (presort != NULL)
168  && ((*presort) == LAGraph_TriangleCount_AutoSort) ;
169  if (auto_sort && method_can_use_presort)
170  {
171  LG_ASSERT_MSG (Degree != NULL,
172  LAGRAPH_NOT_CACHED, "G->out_degree is required") ;
173  }
174 
175  //--------------------------------------------------------------------------
176  // initializations
177  //--------------------------------------------------------------------------
178 
179  GrB_Index n ;
180  GRB_TRY (GrB_Matrix_nrows (&n, A)) ;
181  GRB_TRY (GrB_Matrix_new (&C, GrB_INT64, n, n)) ;
182  #if LAGRAPH_SUITESPARSE
183  GrB_Semiring semiring = GxB_PLUS_PAIR_INT64 ;
184  #else
185  GrB_Semiring semiring = LAGraph_plus_one_int64 ;
186  #endif
187  GrB_Monoid monoid = GrB_PLUS_MONOID_INT64 ;
188 
189  //--------------------------------------------------------------------------
190  // heuristic sort rule
191  //--------------------------------------------------------------------------
192 
193  if (auto_sort)
194  {
195  // auto selection of sorting method
196  (*presort) = LAGraph_TriangleCount_NoSort ; // default is not to sort
197 
198  if (method_can_use_presort)
199  {
200  // This rule is very similar to Scott Beamer's rule in the GAP TC
201  // benchmark, except that it is extended to handle the ascending
202  // sort needed by methods 3 and 5. It also uses a stricter rule,
203  // since the performance of triangle counting in SuiteSparse:
204  // GraphBLAS is less sensitive to the sorting as compared to the
205  // GAP algorithm. This is because the dot products in SuiteSparse:
206  // GraphBLAS use binary search if one vector is very sparse
207  // compared to the other. As a result, SuiteSparse:GraphBLAS needs
208  // the sort for fewer matrices, as compared to the GAP algorithm.
209 
210  // With this rule, the GAP-kron and GAP-twitter matrices are
211  // sorted, and the others remain unsorted. With the rule in the
212  // GAP tc.cc benchmark, GAP-kron and GAP-twitter are sorted, and so
213  // is GAP-web, but GAP-web is not sorted here.
214 
215  #define NSAMPLES 1000
216  GrB_Index nvals ;
217  GRB_TRY (GrB_Matrix_nvals (&nvals, A)) ;
218  if (n > NSAMPLES && ((double) nvals / ((double) n)) >= 10)
219  {
220  // estimate the mean and median degrees
221  double mean, median ;
222  LG_TRY (LAGr_SampleDegree (&mean, &median,
223  G, true, NSAMPLES, n, msg)) ;
224  // sort if the average degree is very high vs the median
225  if (mean > 4 * median)
226  {
227  switch (method)
228  {
229  case LAGraph_TriangleCount_Sandia:
230  // 3:sum (sum ((L * L) .* L))
231  (*presort) = LAGraph_TriangleCount_Ascending ;
232  break ;
233  case LAGraph_TriangleCount_Sandia2:
234  // 4: sum (sum ((U * U) .* U))
235  (*presort) = LAGraph_TriangleCount_Descending ;
236  break ;
237  default:
238  case LAGraph_TriangleCount_SandiaDot:
239  // 5: sum (sum ((L * U') .* L))
240  (*presort) = LAGraph_TriangleCount_Ascending ;
241  break ;
242  case LAGraph_TriangleCount_SandiaDot2:
243  // 6: sum (sum ((U * L') .* U))
244  (*presort) = LAGraph_TriangleCount_Descending ;
245  break ;
246  }
247  }
248  }
249  }
250  }
251 
252  //--------------------------------------------------------------------------
253  // sort the input matrix, if requested
254  //--------------------------------------------------------------------------
255 
256  if (presort != NULL && (*presort) != LAGraph_TriangleCount_NoSort)
257  {
258  // P = permutation that sorts the rows by their degree
259  LG_TRY (LAGr_SortByDegree (&P, G, true,
260  (*presort) == LAGraph_TriangleCount_Ascending, msg)) ;
261 
262  // T = A (P,P) and typecast to boolean
263  GRB_TRY (GrB_Matrix_new (&T, GrB_BOOL, n, n)) ;
264  GRB_TRY (GrB_extract (T, NULL, NULL, A, (GrB_Index *) P, n,
265  (GrB_Index *) P, n, NULL)) ;
266  A = T ;
267 
268  // free workspace
269  LG_TRY (LAGraph_Free ((void **) &P, NULL)) ;
270  }
271 
272  //--------------------------------------------------------------------------
273  // count triangles
274  //--------------------------------------------------------------------------
275 
276  int64_t ntri ;
277 
278  switch (method)
279  {
280 
281  case LAGraph_TriangleCount_Burkhardt: // 1: sum (sum ((A^2) .* A)) / 6
282 
283  GRB_TRY (GrB_mxm (C, A, NULL, semiring, A, A, GrB_DESC_S)) ;
284  GRB_TRY (GrB_reduce (&ntri, NULL, monoid, C, NULL)) ;
285  ntri /= 6 ;
286  break ;
287 
288  case LAGraph_TriangleCount_Cohen: // 2: sum (sum ((L * U) .* A)) / 2
289 
290  LG_TRY (tricount_prep (&L, &U, A, msg)) ;
291  GRB_TRY (GrB_mxm (C, A, NULL, semiring, L, U, GrB_DESC_S)) ;
292  GRB_TRY (GrB_reduce (&ntri, NULL, monoid, C, NULL)) ;
293  ntri /= 2 ;
294  break ;
295 
296  case LAGraph_TriangleCount_Sandia: // 3: sum (sum ((L * L) .* L))
297 
298  // using the masked saxpy3 method
299  LG_TRY (tricount_prep (&L, NULL, A, msg)) ;
300  GRB_TRY (GrB_mxm (C, L, NULL, semiring, L, L, GrB_DESC_S)) ;
301  GRB_TRY (GrB_reduce (&ntri, NULL, monoid, C, NULL)) ;
302  break ;
303 
304  case LAGraph_TriangleCount_Sandia2: // 4: sum (sum ((U * U) .* U))
305 
306  // using the masked saxpy3 method
307  LG_TRY (tricount_prep (NULL, &U, A, msg)) ;
308  GRB_TRY (GrB_mxm (C, U, NULL, semiring, U, U, GrB_DESC_S)) ;
309  GRB_TRY (GrB_reduce (&ntri, NULL, monoid, C, NULL)) ;
310  break ;
311 
312  default:
313  case LAGraph_TriangleCount_SandiaDot: // 5: sum (sum ((L * U') .* L))
314 
315  // This tends to be the fastest method for most large matrices, but
316  // the SandiaDot2 method is also very fast.
317 
318  // using the masked dot product
319  LG_TRY (tricount_prep (&L, &U, A, msg)) ;
320  GRB_TRY (GrB_mxm (C, L, NULL, semiring, L, U, GrB_DESC_ST1)) ;
321  GRB_TRY (GrB_reduce (&ntri, NULL, monoid, C, NULL)) ;
322  break ;
323 
324  case LAGraph_TriangleCount_SandiaDot2: // 6: sum (sum ((U * L') .* U))
325 
326  // using the masked dot product
327  LG_TRY (tricount_prep (&L, &U, A, msg)) ;
328  GRB_TRY (GrB_mxm (C, U, NULL, semiring, U, L, GrB_DESC_ST1)) ;
329  GRB_TRY (GrB_reduce (&ntri, NULL, monoid, C, NULL)) ;
330  break ;
331  }
332 
333  //--------------------------------------------------------------------------
334  // return result
335  //--------------------------------------------------------------------------
336 
337  LG_FREE_ALL ;
338  (*ntriangles) = (uint64_t) ntri ;
339  return (GrB_SUCCESS) ;
340 }
