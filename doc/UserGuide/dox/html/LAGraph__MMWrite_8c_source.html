46 // Print a double value to the file, using the shortest format that ensures the
47 // value is written precisely. Returns true if successful, false if an I/O
48 // error occurred.
49 
50 static bool print_double
51 (
52  FILE *f, // file to print to
53  double x // value to print
54 )
55 {
56 
57  char s [MAXLINE], *p ;
58  int64_t i, dest = 0, src = 0 ;
59  int width, ok ;
60 
61  //--------------------------------------------------------------------------
62  // handle Inf and NaN
63  //--------------------------------------------------------------------------
64 
65  if (isnan (x))
66  {
67  return (fprintf (f, "nan") > 0) ;
68  }
69  if (isinf (x))
70  {
71  return (fprintf (f, (x < 0) ? "-inf" : "inf") > 0) ;
72  }
73 
74  //--------------------------------------------------------------------------
75  // find the smallest acceptable precision
76  //--------------------------------------------------------------------------
77 
78  for (width = 6 ; width < 20 ; width++)
79  {
80  double y ;
81  sprintf (s, "%.*g", width, x) ;
82  sscanf (s, "%lg", &y) ;
83  if (x == y) break ;
84  }
85 
86  //--------------------------------------------------------------------------
87  // shorten the string
88  //--------------------------------------------------------------------------
89 
90  // change "e+0" to "e", change "e+" to "e", and change "e-0" to "e-"
91  for (i = 0 ; i < MAXLINE && s [i] != '\0' ; i++)
92  {
93  if (s [i] == 'e')
94  {
95  if (s [i+1] == '+')
96  {
97  dest = i+1 ;
98  if (s [i+2] == '0')
99  {
100  // delete characters s[i+1] and s[i+2]
101  src = i+3 ;
102  }
103  else
104  {
105  // delete characters s[i+1]
106  src = i+2 ;
107  }
108  }
109  else if (s [i+1] == '-')
110  {
111  dest = i+2 ;
112  if (s [i+2] == '0')
113  {
114  // delete character s[i+2]
115  src = i+3 ;
116  }
117  else
118  {
119  // no change
120  break ;
121  }
122  }
123  while (s [src] != '\0')
124  {
125  s [dest++] = s [src++] ;
126  }
127  s [dest] = '\0' ;
128  break ;
129  }
130  }
131 
132  // delete the leading "0" if present and not necessary
133  p = s ;
134  s [MAXLINE-1] = '\0' ;
135  i = strlen (s) ;
136  if (i > 2 && s [0] == '0' && s [1] == '.')
137  {
138  // change "0.x" to ".x"
139  p = s + 1 ;
140  }
141  else if (i > 3 && s [0] == '-' && s [1] == '0' && s [2] == '.')
142  {
143  // change "-0.x" to "-.x"
144  s [1] = '-' ;
145  p = s + 1 ;
146  }
147 
148 #if 0
149  // double-check
150  i = sscanf (p, "%lg", &z) ;
151  if (i != 1 || y != z)
152  {
153  // oops! something went wrong in the "e+0" edit, above.
154  // this "cannot" happen
155  sprintf (s, "%.*g", width, x) ;
156  p = s ;
157  }
158 #endif
159 
160  //--------------------------------------------------------------------------
161  // print the value to the file
162  //--------------------------------------------------------------------------
163 
164  return (fprintf (f, "%s", p) > 0) ;
165 }
