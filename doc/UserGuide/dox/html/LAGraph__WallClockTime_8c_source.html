14 // Unlike all other LAGraph functions, this function does not return an error
15 // code as an int, nor does it have a char *msg parameter for error messages.
16 // It simply returns the current wall clock time, as a double value, indicating
17 // the amount of time passed in seconds since some fixed point in the past.
18 
19 // Example usage:
20 
21 /*
22  double t1 = LAGraph_WallClockTime ( ) ;
23 
24  // ... do stuff
25  double t2 = LAGraph_WallClockTime ( ) ;
26  printf ("time to 'do stuff' : %g (seconds)\n', t2 - t1) ;
27  // ... more stuff
28  double t3 = LAGraph_WallClockTime ( ) ;
29  printf ("time to 'do stuff' and 'more stuff': %g (seconds)\n', t3 - t1) ;

45 double LAGraph_WallClockTime (void)
46 {
47  double t_wallclock = 0 ;
48 
49  #if defined ( _OPENMP )
50 
51  // OpenMP is available; use the OpenMP timer function
52  t_wallclock = omp_get_wtime ( ) ;
53 
54  #elif defined ( __linux__ )
55 
56  // Linux has a very low resolution clock() function, so use the high
57  // resolution clock_gettime instead. May require -lrt
58  struct timespec t ;
59  int e = clock_gettime (CLOCK_MONOTONIC, &t) ;
60  if (e == 0)
61  {
62  t_wallclock = (double) t.tv_sec + 1e-9 * ((double) t.tv_nsec) ;
63  }
64 
65  #elif defined ( __MACH__ )
66 
67  // Mac OSX
68  clock_serv_t cclock ;
69  mach_timespec_t t ;
70  host_get_clock_service (mach_host_self ( ), SYSTEM_CLOCK, &cclock) ;
71  clock_get_time (cclock, &t) ;
72  mach_port_deallocate (mach_task_self ( ), cclock) ;
73  t_wallclock = (double) t.tv_sec + 1e-9 * ((double) t.tv_nsec) ;
74 
75  #else
76 
77  // The ANSI C11 clock() function is used instead. This gives the
78  // processor time, not the wallclock time, and it might have low
79  // resolution. It returns the time since some unspecified fixed time
80  // in the past, as a clock_t integer. The clock ticks per second are
81  // given by CLOCKS_PER_SEC. In Mac OSX this is a very high resolution
82  // clock, and clock ( ) is faster than clock_get_time (...) ;
83  clock_t t = clock ( ) ;
84  t_wallclock = ((double) t) / ((double) CLOCKS_PER_SEC) ;
85 
86  #endif
87 
88  return (t_wallclock) ;
89 }
