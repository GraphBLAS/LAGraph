14 // LAGr_SortByDegree computes a permutation vector P that sorts a graph
15 // by degree (either row or column degree of its adjacency matrix A).
16 // If G is undirected, or if G is directed but is known to have a symmetric
17 // adjacency matrix, then G->out_degree is used (and byout is ignored).
18 // Otherwise, if G->out_degree is used if byout is true, and G->in_degree is
19 // used if byout is false.
20 
21 // G->out_degree or G->in_degree must first be computed. An error is returned
22 // if the required degree vector has not yet been computed. See
23 // LAGraph_Cached_OutDegree and LAGraph_Cached_InDegree.
24 
25 // The permutation is in ascending order of degree if ascending is true, and
26 // in descending order otherwise.
27 
28 // Ties are broken by the node id, so the sort is always predicable. Lower
29 // numbered rows/columns always appear before higher ones, if they have the
30 // same degree.
31 
32 // The output is a permutation P where P [k] = i if row i is the kth row in
33 // the permutation (or P [k] = j if column j is the kth column in the
34 // permutation, with byout false).
35

49 
50 int LAGr_SortByDegree
51 (
52  // output:
53  int64_t **P_handle, // permutation vector of size n
54  // input:
55  const LAGraph_Graph G, // graph of n nodes
56  bool byout, // if true, sort G->out_degree, else G->in_degree
57  bool ascending, // sort in ascending or descending order
58  char *msg
59 )
60 {
61 
62  //--------------------------------------------------------------------------
63  // check inputs
64  //--------------------------------------------------------------------------
65 
66  LG_CLEAR_MSG ;
67  int64_t *P = NULL ;
68  int64_t *W = NULL ;
69  int64_t *D = NULL ;
70  LG_ASSERT_MSG (P_handle != NULL, GrB_NULL_POINTER, "&P != NULL") ;
71  (*P_handle) = NULL ;
72  LG_TRY (LAGraph_CheckGraph (G, msg)) ;
73 
74  GrB_Vector Degree ;
75 
76  if (G->kind == LAGraph_ADJACENCY_UNDIRECTED ||
77  (G->kind == LAGraph_ADJACENCY_DIRECTED &&
78  G->is_symmetric_structure == LAGraph_TRUE))
79  {
80  // the structure of A is known to be symmetric
81  Degree = G->out_degree ;
82  }
83  else
84  {
85  // A is not known to be symmetric
86  Degree = (byout) ? G->out_degree : G->in_degree ;
87  }
88 
89  LG_ASSERT_MSG (Degree != NULL, LAGRAPH_NOT_CACHED, "degree unknown") ;
90 
91  //--------------------------------------------------------------------------
92  // decide how many threads to use
93  //--------------------------------------------------------------------------
94 
95  GrB_Index n ;
96  GRB_TRY (GrB_Vector_size (&n, Degree)) ;
97 
98  #define CHUNK (64*1024)
99  int nthreads = LG_nthreads_outer * LG_nthreads_inner ;
100  nthreads = LAGRAPH_MIN (nthreads, n/CHUNK) ;
101  nthreads = LAGRAPH_MAX (nthreads, 1) ;
102 
103  //--------------------------------------------------------------------------
104  // allocate result and workspace
105  //--------------------------------------------------------------------------
106 
107  LG_TRY (LAGraph_Malloc ((void **) &P, n, sizeof (int64_t), msg)) ;
108  LG_TRY (LAGraph_Malloc ((void **) &D, n, sizeof (int64_t), msg)) ;
109  LG_TRY (LAGraph_Malloc ((void **) &W, 2*n, sizeof (int64_t), msg)) ;
110  int64_t *W0 = W ;
111  int64_t *W1 = W + n ;
112 
113  //--------------------------------------------------------------------------
114  // construct the pair [D,P] to sort
115  //--------------------------------------------------------------------------
116 
117  #pragma omp parallel for num_threads(nthreads) schedule(static)
118  for (int64_t k = 0 ; k < n ; k++)
119  {
120  D [k] = 0 ;
121  P [k] = k ;
122  }
123 
124  // extract the degrees
125  GrB_Index nvals = n ;
126  GRB_TRY (GrB_Vector_extractTuples ((GrB_Index *) W0, W1, &nvals, Degree)) ;
127 
128  if (ascending)
129  {
130  // sort [D,P] in ascending order of degree, tie-breaking on P
131  #pragma omp parallel for num_threads(nthreads) schedule(static)
132  for (int64_t k = 0 ; k < nvals ; k++)
133  {
134  D [W0 [k]] = W1 [k] ;
135  }
136  }
137  else
138  {
139  // sort [D,P] in descending order of degree, tie-breaking on P
140  #pragma omp parallel for num_threads(nthreads) schedule(static)
141  for (int64_t k = 0 ; k < nvals ; k++)
142  {
143  D [W0 [k]] = -W1 [k] ;
144  }
145  }
146 
147  LG_TRY (LAGraph_Free ((void **) &W, NULL)) ;
148 
149  //--------------------------------------------------------------------------
150  // sort by degrees, with ties by node id
151  //--------------------------------------------------------------------------
152 
153  LG_TRY (LG_msort2 (D, P, n, msg)) ;
154 
155  //--------------------------------------------------------------------------
156  // free workspace and return result
157  //--------------------------------------------------------------------------
158 
159  LG_FREE_WORK ;
160  (*P_handle) = P ;
161  return (GrB_SUCCESS) ;
162 }
